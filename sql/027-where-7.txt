7. Display the employee name, job, and salary for all
        employees whose job is CLERK or ANALYST and their salary is
        not equal to 1000 or 5000.

select ename, job, sal
FROM emp
WHERE
(job = 'CLERK' OR job = 'ANALYST')
AND
(sal <> 1000 OR sal <> 5000)
;
